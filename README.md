## Install

```bash
npm install --save use-route
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'use-route'
import 'use-route/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
